Compute the value of each of the following.  We encourage you to keep it in exponential form as long as you can, for practice. \textbf{Remember to show your work.}
\begin{qparts}
    \item $5^3$
    \item $2^3\cdot 2^2$

    \item $5^0$
    \item $(2^4)^2$
    \item $2^{(4^2)}$ (use a calculator for this one if you need to!)
    \item $16^{\frac 12}$
    \item $16^{-2}$

  a) $5^3$=$5\cdot 5\cdot 5$=$125$\\b) $2^3\cdot 2^2=2^5=32$\\c) $\frac{3^6}{3^3}=3^3=27$\\d) $5^0=1$\\e) $(2^4)^2=2^8=256$\\f) $2^(4^2)=2^{16}=65536$\\g) $16^{\frac 12}=4$\\h) $16^{-2}={\frac 1{16^2}}={\frac {1}{256}}$
\end{solution}

\subsection*{\probnum Logarithms [16 points]}
Compute the value of each of the following.  If the answer does not come out cleanly and results in an infinite decimal expansion, write ``\textit{too hard to compute}." If the result is not defined, write ``\textit{undefined}." For any question referencing $x$ or $y,$ let $\log_5(x)=2.6$ and $\log_5(y)=5.2.$ \textbf{Remember to show your work.}
\begin{qparts}
    \item $\log_2{64}$
    \item $\log_7{7}$
    \item $\log_7{1}$
    \item $\log_7{0}$
    \item $\log_5(x+y)$
    \item $\log_5(xy)$
    \item $\log_2(x)$
    \item $\log_5(x^{25})$
\end{qparts}

\begin{solution}
  a) $\log_2{64}=6$
  \\b) $\log_7{7}=1$
  \\c) $\log_7{1}=0$
  \\d) $\log_7{0}=$undefined
  \\e) if we are allowed to use a calculator, then $\log_5(x)=2.6\;x=65.66\;\log_5(y)=5.2\;y=4311.66\;\log_5(x+y)=5.21$ or we can say $\log_5(y)$ is much bigger than $\log_5(x)$ thus $\log_5(x+y)$ is pretty close to $\log_5(y)$ which is 5.2, or it is too hard to compute.
  \\f) $\log_5(xy)=\log_5(x)+\log_5(y)=2.6+5.2=7.8$
  \\g) if we are allowed to use a calculator, then $\log_2(x)=\log_2(65.66)=6.0$ or we can use change of base, $\log_2(x)={\frac {\log_5(x)}{\log_5(2)}}={\frac {2.6}{0.43}}=6.0$
  \\h) $\log_5(x^{25})=25\cdot \log_5(x)=25\cdot 2.6=65$
\end{solution}
